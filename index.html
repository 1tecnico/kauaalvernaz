import tkinter as tk from datetime import datetime # Função para enviar mensagens def enviar_mensagem(): mensagem = entrada.get() if mensagem.strip() != "": hora = datetime.now().strftime("%H:%M") texto_chat.config(state='normal') texto_chat.insert(tk.END, f"Você ({hora}): {mensagem}\n") texto_chat.insert(tk.END, f"Bot ({hora}): {responder(mensagem)}\n\n") texto_chat.config(state='disabled') entrada.delete(0, tk.END) texto_chat.see(tk.END) # Função de resposta do "bot" def responder(mensagem): mensagem = mensagem.lower() if "oi" in mensagem or "olá" in mensagem: return "Olá! Como posso ajudar?" elif "tudo bem" in mensagem: return "Tudo ótimo! E com você?" elif "obrigado" in mensagem: return "De nada 😊" else: return "Desculpe, ainda estou aprendendo a responder isso." # Criar janela principal janela = tk.Tk() janela.title("Chat Simples") janela.geometry("400x500") # Área de chat texto_chat = tk.Text(janela, state='disabled', wrap='word') texto_chat.pack(padx=10, pady=10, fill=tk.BOTH, expand=True) # Campo de entrada e botão frame_entrada = tk.Frame(janela) frame_entrada.pack(padx=10, pady=10, fill=tk.X) entrada = tk.Entry(frame_entrada) entrada.pack(side=tk.LEFT, fill=tk.X, expand=True, padx=(0, 10)) entrada.bind("", lambda event: enviar_mensagem()) botao_enviar = tk.Button(frame_entrada, text="Enviar", command=enviar_mensagem) botao_enviar.pack(side=tk.RIGHT) # Iniciar o aplicativo janela.mainloop() Chat Simples
